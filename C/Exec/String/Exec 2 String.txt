/*
    2) Escrever um programa em C que leia uma mensagem e conte a quantidade de caracteres brancos.
*/

#include <stdio.h>
#include <stdlib.h>
//CONSTANTES
#define TAM 100

void cont_branco(char *MSG, int *contador);

void cont_branco(char *MSG, int *contador){
    int i = 0;
    for(i = 0; MSG[i] != '\0'; i++){
    if(MSG[i] == ' ')
    (*contador)++;
    }
}

int main(){
    char MSG[TAM];
    int contador = 0;
    printf("Digite uma mensagem: \n"); scanf("%[^\n]", MSG);
    cont_branco(MSG, &contador);
    printf("Quantidade de espacos em branco Ã© de: %i", contador);
    printf("\n");
    return 0;
}
